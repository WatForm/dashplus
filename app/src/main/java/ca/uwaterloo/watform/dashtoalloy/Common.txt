    // p_stateName_paramSig

    public ExprVar paramVar() {
        return ExprHelper.createVar(
        	     DashStrings.pName
        	     + DashStrings.alloySep
        	     + this.stateName);
    }

    // p_stateName:paramSig
    public Decl paramDecl() {
        return (Decl) 
        	new DeclExt(
        			// name
        			paramVar(),
        			// type 
        			ExprHelper.createVar(this.paramSig),
        			// isVar
        			true 
        		);
    }

        // returns the primed variables in an exp (but w/o the primes)
    public List<DashRef> primedDashRef(AlloyExpr exp) {
        List<DashRef> drs = new collectDashRef().visit(exp);
        List<DashRef> o = new ArrayList<DashRef>();
        String v;
        List<DashExpr> paramValues;
        for (DashRef e: drs) {
            v = e.name;
            paramValues = e.paramValues;
            if (hasPrime(v)) {
                o.add(new VarDashRef(removePrime(v), paramValues));
            }
        }
        return o;
    }
