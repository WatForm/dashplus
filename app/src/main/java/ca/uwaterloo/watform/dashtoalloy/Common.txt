

        // returns the primed variables in an exp (but w/o the primes)
    public List<DashRef> primedDashRef(AlloyExpr exp) {
        List<DashRef> drs = new collectDashRef().visit(exp);
        List<DashRef> o = new ArrayList<DashRef>();
        String v;
        List<DashExpr> paramValues;
        for (DashRef e: drs) {
            v = e.name;
            paramValues = e.paramValues;
            if (hasPrime(v)) {
                o.add(new VarDashRef(removePrime(v), paramValues));
            }
        }
        return o;
    }
